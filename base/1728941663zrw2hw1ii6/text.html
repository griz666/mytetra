Для автомапускаправил после создания своих докером надо их собрать в скрипт и подгружать через systemd

iptables.service


[Unit]

Description=IPTables rules

After=network.target


[Service]

Type=oneshot

ExecStart=/etc/network/firewall/ipt.sh start

RemainAfterExit=yes


[Install]

WantedBy=multi-user.target
